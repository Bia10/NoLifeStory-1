project(NoLifeStory)
cmake_minimum_required(VERSION 2.8.12 FATAL_ERROR)
include(cmake/common.cmake)
enable_cxx11()

add_subdirectory(nx)
add_subdirectory(client)

option(BUILD_WZTONX "Whether to build wztonx or not" ON)
if(BUILD_WZTONX)
  message(STATUS "Adding wztonx to build")
  add_subdirectory(wztonx)
endif()

option(ALL_WARNINGS "Notify compiler to be an arse" ON)
if(ALL_WARNINGS)
  message(STATUS "All warnings enabled")
  if(UNIX)
    add_compile_options(-Wall -Wpedantic)
  endif()
endif()
